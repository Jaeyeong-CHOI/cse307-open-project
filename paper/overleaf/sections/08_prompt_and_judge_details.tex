\subsection{L2: Syntax Inversion}

\begin{figure}[!t]
\small
\begin{tcolorbox}[colback=orange!4, colframe=orange!60,
  title=\textbf{L2 grammar specification (complete)},
  fonttitle=\small\bfseries, boxrule=0.6pt]
\textbf{This is NOT Python.} All Python keywords are FORBIDDEN.\\[4pt]
\texttt{:define name [args] ->}\quad (replaces \texttt{def name(args):})\\
\texttt{<= value}\quad (replaces \texttt{return value})\\
\texttt{value ?cond: res ;else: alt ;end}\quad (replaces \texttt{if})\\
\texttt{loop var over iterable \{ body \}}\quad (replaces \texttt{for})\\
\texttt{;;}\quad closes function block
\end{tcolorbox}
\caption{L2 grammar specification as given to the model. Models must produce code using exclusively these constructs. Result: all models fully adopted the syntax (SIR=0.0), showing LLMs readily follow explicit structural rules.}

\end{figure}

\subsection{L3: Semantic Inversion (Explicit)}

\begin{figure}[!t]
\small
\begin{tcolorbox}[colback=red!3, colframe=red!50,
  title=\textbf{L3 system prompt --- explicit semantic rule},
  fonttitle=\small\bfseries, boxrule=0.6pt]
\textit{``Follow the L3 language rules exactly. If-blocks run when condition is \textbf{FALSE}.''}
\end{tcolorbox}
\vspace{3pt}
\begin{tcolorbox}[colback=red!5, colframe=red!60,
  title=\textbf{L3 user prompt (complete)},
  fonttitle=\small\bfseries, boxrule=0.6pt]
\textbf{L3 rule:} Syntax is identical to Python, but:\\
RULE: Indented block executes when condition is \textbf{FALSE}.\\
RULE: Else-block executes when condition is \textbf{TRUE}.\\[4pt]
\textit{Example:}
\begin{verbatim}
if x > 5:
    print("small")  # runs when NOT x > 5
else:
    print("large")  # runs when x > 5
\end{verbatim}
\vspace{2pt}
\textbf{Task:} Write \texttt{fib(n)} in L3. Intended logic: if $n \le 1$ return $n$; else return \texttt{fib(n-1)+fib(n-2)}.\\
Think carefully: your if-block runs when condition is FALSE.
\end{tcolorbox}
\caption{L3 prompts. The rule is stated explicitly in the system message. Weaker models (gpt-4o-mini) still fail (PPR=1.0), while stronger models succeed---revealing a capability threshold for semantic meta-rule following.}
\label{fig:l3_prompts}
\end{figure}

\subsection{L4: Semantic Inversion (Implicit)}

\begin{figure}[!t]
\small
\begin{tcolorbox}[colback=purple!4, colframe=purple!55,
  title=\textbf{L4 user prompt (example-only delivery, no explicit rule)},
  fonttitle=\small\bfseries, boxrule=0.6pt]
Here are some examples in our programming language:\\[4pt]
\textbf{Example 1 --- max function:}
\begin{verbatim}
def max_val(a, b):
    if a > b:
        return b   % smaller when a > b
    return a
\end{verbatim}
Output: \texttt{max\_val(3,5)=3},\ \texttt{max\_val(7,2)=2}\\[4pt]
\textbf{Example 2 --- even check:}
\begin{verbatim}
def is_even(n):
    if n % 2 == 0:
        return False
    return True
\end{verbatim}
Output: \texttt{is\_even(4)=False},\ \texttt{is\_even(3)=True}\\[4pt]
\textbf{Task:} Write \texttt{fib(n)} in this language.
Target: \texttt{fib(0)=0}, \texttt{fib(1)=1}, \texttt{fib(n)=fib(n-1)+fib(n-2)}.\\
Print \texttt{fib(0)} through \texttt{fib(6)}.
\end{tcolorbox}
\caption{L4 prompt (no system message). The semantic inversion rule is not stated; models must infer it from worked examples. Even gpt-4o, which succeeds at L3, defaults to canonical Python fib---exhibiting \emph{pattern blindness}.}
\label{fig:l4_prompt}
\end{figure}

\subsection{Evaluation Metrics}
\begin{itemize}
  \item \textbf{KLR} (Keyword Leakage Rate): fraction of alias slots where original Python keyword was used. ACR $= 1 - \text{KLR}$ is a derived quantity.
  \item \textbf{PSS} (Partial Structural Score): weighted sub-component credit (alias: 0.4, parse: 0.25, skeleton: 0.25, format: 0.1).
  \item \textbf{PPR} (Python Prior Rate, L3/L4): fraction of outputs using Python-style condition without inversion.
\end{itemize}

\subsection{Judge Protocol (L1)}
\begin{enumerate}
  \item Alias compliance: check that alias is present and original token is absent.
  \item Normalization: replace aliases $\to$ Python keywords via word-boundary regex.
  \item Parse check: \texttt{ast.parse()} on normalized code.
  \item Skeleton check: verify \texttt{fib}, base case, recursion, and print loop.
\end{enumerate}

\subsection{Judge Protocol (L3/L4)}
\begin{itemize}
  \item \textbf{Python-prior}: model writes \texttt{if n <= 1: return n} (non-inverted) $\Rightarrow$ PPR=1 (fail).
  \item \textbf{L3-correct}: model writes \texttt{if n > 1: return n} or equivalent $\Rightarrow$ PPR=0 (pass).
\end{itemize}
