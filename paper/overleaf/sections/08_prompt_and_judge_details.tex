\subsection{L2: Syntax Inversion}

Grammar specification given to model:

\begin{mdframed}[backgroundcolor=gray!8, hidealllines=true, innerleftmargin=6pt, innerrightmargin=6pt, innertopmargin=4pt, innerbottommargin=4pt]
\small
\textbf{This is NOT Python.} All standard Python keywords (\texttt{def}, \texttt{return}, \texttt{if}, \texttt{for}, \texttt{in}) are FORBIDDEN.

Grammar:\\
\texttt{:define name [args] ->}\quad (not \texttt{def name(args):})\\
\texttt{<= value}\quad (not \texttt{return value})\\
\texttt{value ?cond: res ;else: alt ;end}\quad (not \texttt{if})\\
\texttt{loop var over iter \{ body \}}\quad (not \texttt{for})\\
\texttt{;;}\quad closes function block
\end{mdframed}

\subsection{L3: Semantic Inversion (Explicit)}

\subsubsection{System prompt}
\begin{quote}
\small
\textit{Follow the L3 language rules exactly. If-blocks run when condition is FALSE.}
\end{quote}

\subsubsection{User prompt}

\begin{mdframed}[backgroundcolor=gray!8, hidealllines=true, innerleftmargin=6pt, innerrightmargin=6pt, innertopmargin=4pt, innerbottommargin=4pt]
\small
\textbf{L3 language rule (semantic inversion):}\\
The syntax is identical to Python, but:\\
RULE: The indented block of \texttt{if} executes when condition is \textbf{FALSE}.\\
RULE: \texttt{else} executes when condition is \textbf{TRUE}.

\medskip\noindent Example:
\begin{verbatim}
if x > 5:
    print("small")  # runs when NOT x > 5
else:
    print("large")  # runs when x > 5
\end{verbatim}

\medskip\noindent\textbf{Task:} Write fib function in L3. Intended logic: if $n \le 1$, return $n$; else return \texttt{fib(n-1)+fib(n-2)}.\\
Think carefully about how to invert your condition.\\
Loop $i = 0..6$, print \texttt{(i, fib(i))}.
\end{mdframed}

\subsection{L4: Semantic Inversion (Implicit)}

User prompt only (no system message):

\begin{mdframed}[backgroundcolor=gray!8, hidealllines=true, innerleftmargin=6pt, innerrightmargin=6pt, innertopmargin=4pt, innerbottommargin=4pt]
\small
Here are some examples in our programming language:

\medskip\noindent\textbf{Example 1} -- max function:
\begin{verbatim}
def max_val(a, b):
    if a > b:
    return b  % returns smaller when a>b
    return a
\end{verbatim}
Output: \texttt{max\_val(3,5)=3}, \texttt{max\_val(7,2)=2}

\medskip\noindent\textbf{Example 2} -- even check:
\begin{verbatim}
def is_even(n):
    if n % 2 == 0:
        return False
    return True
\end{verbatim}
Output: \texttt{is\_even(4)=False}, \texttt{is\_even(3)=True}

\medskip\noindent\textbf{Task:} Write Fibonacci in this language. Target: \texttt{fib(0)=0}, \texttt{fib(1)=1}, \texttt{fib(n)=fib(n-1)+fib(n-2)}.\\
Print \texttt{fib(0)} through \texttt{fib(6)}.
\end{mdframed}

\subsection{Judge Protocol (L1)}

\begin{enumerate}
  \item \textbf{Alias compliance}: check that each required alias is present and original token is absent.
  \item \textbf{Normalization}: replace aliases back to Python via word-boundary regex.
  \item \textbf{Parse check}: \texttt{ast.parse()} on normalized code.
  \item \textbf{Skeleton}: verify \texttt{fib}, base case, recursion, and print loop.
\end{enumerate}

Metrics (all continuous):
\begin{align*}
\text{KLR} &= \frac{\text{\# leaked token slots}}{\text{\# active alias slots}} \\
\text{ACR} &= 1 - \text{KLR} \\
\text{PSS} &= 0.4 \cdot \text{ACR} + 0.25 \cdot \text{parse} + 0.25 \cdot \text{skel} + 0.1 \cdot \text{loop} \\
\text{PIR} &= \frac{\text{KLR}}{\text{KLR} + \text{ACR}}
\end{align*}

\subsection{Judge Protocol (L3/L4)}

\begin{itemize}
  \item \textbf{Python-prior}: model writes \texttt{if n <= 1: return n} (no inversion) $\Rightarrow$ fail.
  \item \textbf{L3-correct}: model writes \texttt{if n > 1: return n} or equivalent $\Rightarrow$ pass.
  \item $\text{PPR} = $ fraction of samples using Python-prior pattern.
\end{itemize}
