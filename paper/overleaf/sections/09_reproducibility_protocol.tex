\section{Complete Experiment Log and Reproducibility}

\subsection{All experiments conducted (in order)}

\begin{enumerate}
  \item \textbf{L1 baseline, n=120 $\times$ 4 models (2026-03-23)}\\
    Models: gpt-4o-mini, gpt-4.1-mini, gpt-4.1-nano, gpt-5.4-mini\\
    Result files: \texttt{prompt-batch-v1-120.\{model\}.2026-03-23.json}

  \item \textbf{L1 context-pack ablation (2026-03-23)}\\
    Model: gpt-4o-mini, CONTEXT\_PACK=1\\
    Result: \texttt{prompt-batch-v1-120-contextpack.gpt4o-mini.2026-03-23.json}\\
    KLR: 0.423 $\to$ 0.260 (39\% reduction)

  \item \textbf{Partial judge re-evaluation (2026-03-24)}\\
    Applied \texttt{partial\_judge.py} to all L1 batch JSONs\\
    Generated \texttt{*.partial\_scores.json} for all 5 conditions

  \item \textbf{L2 experiment, n=5 (2026-03-24)}\\
    Model: gpt-4o-mini, explicit grammar rules\\
    Result: \texttt{L2-batch-quick-gpt4omini.json}\\
    SIR=0.0, L2-adoption=1.0 --- all prompts passed

  \item \textbf{L3 explicit rule, n=5 (2026-03-24)}\\
    Model: gpt-4o-mini, system prompt with semantic inversion rule\\
    Result: \texttt{L3-semantic-batch-gpt4omini.json}\\
    PPR=1.0 --- model completely failed (used Python-prior pattern)

  \item \textbf{L3 explicit rule, n=10 (2026-03-24)}\\
    Models: gpt-4o, gpt-4.1-mini\\
    Results: \texttt{L3-semantic-batch-gpt4o.json}, \texttt{L3-semantic-batch-gpt4.1mini.json}\\
    gpt-4o PPR=0.0, gpt-4.1-mini PPR=0.0 --- both succeeded

  \item \textbf{L4 implicit, n=5 (2026-03-24)}\\
    Model: gpt-4o, no system prompt, only worked examples\\
    Result: \texttt{L4-implicit-batch-gpt4o.json}\\
    Pattern blindness: PPR=0.0 (did not use Python-prior) but noticed inversion=0 --- model ignored examples entirely
\end{enumerate}

\subsection{Key API and runtime settings}
\begin{itemize}
  \item Base URL: \texttt{https://api.openai.com/v1}
  \item Temperature: \texttt{0} (all experiments)
  \item Max tokens: default (no explicit limit)
  \item API key: loaded from AVRTG project env at runtime
  \item Retry policy: none (single attempt per prompt)
  \item Prompt delay: 0.4s between calls
\end{itemize}
